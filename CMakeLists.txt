cmake_minimum_required (VERSION 3.8)
project(AI_LIBRARY VERSION 1.0)
set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT AI_LIBRARY)
set (CMAKE_CXX_STANDARD 20)

set(AI_LIBRARY_VERSION_MAJOR 1 CACHE STRING "major version of ai" FORCE)
set(AI_LIBRARY_VERSION_MINOR 0 CACHE STRING "minor version of ai" FORCE)
set(AI_LIBRARY_VERSION_PATCH 0 CACHE STRING "patch version of ai" FORCE)
set(AI_LIBRARY_VERSION "${AI_LIBRARY_VERSION_MAJOR}.${AI_LIBRARY_VERSION_MINOR}.${AI_LIBRARY_VERSION_PATCH}"
                             CACHE STRING "version of ai" FORCE)

set(AI_LIBRARY_INC_INSTALL_DIR "include/")
set(AI_LIBRARY_RUNTIME_INSTALL_DIR   "bin")
set(AI_LIBRARY_LIBRARY_INSTALL_DIR   "bin")
set(AI_LIBRARY_ARCHIVE_INSTALL_DIR   "lib")
set(AI_LIBRARY_FRAMEWORK_INSTALL_DIR "bin")

set(AI_LIBRARY_CMAKE_CONFIG_INSTALL_DIR "share/ai_library/cmake")
set(AI_LIBRARY_ADDITIONAL_FILES_INSTALL_DIR "share/ai_library")
set(CMAKE_WINDOWS_EXPORT_ALL_SYMBOLS ON)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")

include_directories(${PROJECT_SOURCE_DIR}/include)
file(GLOB_RECURSE SOURCE "source/*.*")
add_library(AI_LIBRARY SHARED ${SOURCE})
